header('Content-Type: text/plain');
$levels = 10;

for($i=0;$i<$levels;$i++){

               echo " ";

       for($j=0;$j<(2*$i)-1;$j++)

               echo "*";

       for($j=0;$j<2*floor($levels-$i);$j++)

               echo " ";

       for($j=0;$j<(2*$i)-1;$j++)

               echo "*";

       for($j=0;$j<2*floor($levels-$i);$j++)

               echo " ";

       for($j=0;$j<(2*$i)-1;$j++)

               echo "*";

echo "\n";

}

?>
